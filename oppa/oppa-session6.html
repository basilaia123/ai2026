-- ❌ ნელი Query (45 წამი) — Full Table Scan

SELECT

t.id, t.amount, t.status, t.created_at,

m.business_name, m.city

FROM transactions t

JOIN merchants m ON t.merchant_id = m.id

WHERE t.status = 'PENDING'

AND t.created_at >= NOW() - INTERVAL '30 days'

ORDER BY t.created_at DESC;

-- EXPLAIN: Seq Scan on transactions (cost=0.00..450000.00 rows=2340000)

Claude Code CLI — Query-ის ანალიზი

გაანალიზე ეს PostgreSQL Query OPPA-ს transactions ბაზაში (10M+ rows):

SELECT t.id, t.amount, t.status, t.created_at, m.business_name
FROM transactions t
JOIN merchants m ON t.merchant_id = m.id
WHERE t.status = 'PENDING'
  AND t.created_at >= NOW() - INTERVAL '30 days'
ORDER BY t.created_at DESC;

import os, logging

from dataclasses import dataclass

from decimal import Decimal

from typing import Optional

import psycopg2

from psycopg2.extras import RealDictCursor

from dotenv import load_dotenv

load_dotenv()

logger = logging.getLogger(__name__)

class PaymentError(Exception): pass

class ValidationError(PaymentError): pass

@dataclass

class PaymentResult:

transaction_id: str

status: str

amount: Decimal

def process_payment(

terminal_id: str,

amount: Decimal,

payment_type: str

) -> Optional[PaymentResult]:

"""Process a PayBox terminal payment.

Args: terminal_id, amount (GEL), payment_type (CASH/CARD/QR)

Returns: PaymentResult | None

Raises: ValidationError, PaymentError

"""

if amount <= 0:

raise ValidationError(f"Invalid amount: {amount}")

try:

conn = psycopg2.connect(os.environ["DATABASE_URL"])

with conn, conn.cursor(cursor_factory=RealDictCursor) as cur:

cur.execute(

"INSERT INTO transactions (terminal_id, amount, type, status)"

"VALUES (%s, %s, %s, 'PENDING') RETURNING id",

(terminal_id, amount, payment_type)

)

row = cur.fetchone()

logger.info("Payment processed: txn_id=%s", row["id"])

return PaymentResult(row["id"], "PENDING", amount)

except psycopg2.Error as e:

logger.error("DB error: %s", e)

raise PaymentError("Database error") from e
